query FindScenes($filter: FindFilterType, $scene_filter: SceneFilterType, $scene_ids: [Int!]) {
        findScenes(filter: $filter, scene_filter: $scene_filter, scene_ids: $scene_ids) {
                count
                filesize
                duration
                scenes {
                        ...Scene
                        __typename
                }
                __typename
        }
}

fragment Scene on Scene {
        id
        title
        code
        details
        director
        urls
        date
        rating100
        organized
        o_counter
        interactive
        interactive_speed
        captions { ...VideoCaption }
        created_at
        updated_at
        last_played_at
        resume_time
        play_duration
        play_count
        play_history
        o_history
        files { ...VideoFile }
        paths { ...ScenePathsType }
        scene_markers { ...SceneMarker }
        galleries { id }
        studio { id }
        groups { ...SceneGroup }
        tags { ...Tag }
        performers { id }
        stash_ids { ...StashID }
        sceneStreams { ...SceneStreamEndpoint }
}
fragment SceneStreamEndpoint on SceneStreamEndpoint {
        url
        mime_type
        label
}
fragment StashID on StashID {
        endpoint
        stash_id
}

        preview

        vtt
        sprite

        caption
}
fragment VideoCaption on VideoCaption {
        language_code
        caption_type
}
fragment SceneGroup on SceneGroup {
        group { ...Group }
        scene_index
}
fragment Tag on Tag {
        id
        name
        description
        aliases
        ignore_auto_tag
        created_at
        updated_at
        favorite

        child_count

        updated_at
